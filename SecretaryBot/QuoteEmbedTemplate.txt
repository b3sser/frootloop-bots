        'QUOTE1',
        'QUOTE2',
    ]    
    
#	NAME_chocolate = [

#        'CHOCOQUOTE2',

    if special == 'oath':
        quote = 'OATHQUOTE'
#    elif special == 'chocolate':
#        quote = random.choice(NAME_chocolate) ODER 'CHOCOLATEQUOTE'
    else:
        quote = random.choice(NAME_quotes_list)
        
    quoteembed = discord.Embed(
        title = 'NAME:',
        description = quote ,
        colour = discord.Colour.COLOUR()
    )
    
    quoteembed.set_thumbnail(url='ICON-URL')
    
    await ctx.send(embed=quoteembed)
